// FIRESTORE SECURITY RULES
// Copy this to Firebase Console > Firestore Database > Rules tab
// Then click "Publish"

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Team Members collection - secure access for owners and employees
    match /team_members/{memberId} {
      // Only owner of the business can create employees
      allow create: if request.auth != null;
      
      // Allow ANYONE to read (needed for employee login authentication)
      allow read: if true;
      
      // Owner can update employee details (status, role, etc)
      allow update: if request.auth != null && request.auth.email == resource.data.createdBy;
      
      // Owner can delete employees
      allow delete: if request.auth != null && request.auth.email == resource.data.createdBy;
    }
    
    // Team invites - existing rules
    match /teamInvites/{inviteId} {
      allow create, read, update: if request.auth != null;
    }
    
    // Fallback rule - authenticated users can read/write other collections
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
